new Vue({
  el: '#exercise',
  data: {
  
  colors: {
  
   highlight: false,
      shrink: true
    },
    

    user: '',
    isVisible: true,
    
    
    myStyle: {
    
    width: '90px',
      height: '100px',
      backgroundColor: 'gray'
    },
    
    
    progressBar: {
    width: '0px',
      backgroundColor: 'darkMagenta'
    }
  },
  
  
  methods: {
  
    startEffect: function() {
    var vm = this;
      setInterval(function() {
      vm.colors.highlight = !vm.colors.highlight;
        vm.colors.shrink = !vm.colors.shrink;
      }, 1000);
    },
    
    startProgress: function() {
    var vm = this;
      var width = 0;
     
    setInterval(function() {
      width = width + 10;
      vm.progressBar.width = width + 'px';
      }, 500);
    }
  }
});
